# VueTest

# Install vue-cli 3
npm install -g @vue-cli

# Create your project
vue create my-name-app

# Enter in your app and run 
cd my-name-app

#if package manager is YARN
yarn serve

#if package manager is NPM
npm run serve

# Set .env file
# Add plugin vuetify
vue add vuetify

# build app
yarn build  
# or
npm run build
